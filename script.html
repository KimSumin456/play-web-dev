배열 고차 함수 (Higher order function)


        console.log("배열 고차 함수 (Higher order function)");

        const numbers = [1, 4, 9, 16, 25];

        numbers.forEach((item, index, self) => {
            console.log(`numbers[${index}] = ${item}`);
        });
        console.log(`numbers: ${numbers}`);
        // numbers: 1,4,9,16,25

        const roots = numbers.map((item, index, self) => {
            return Math.sqrt(item);
        });
        console.log(`roots: ${roots}`);
        // roots: 1,2,3,4,5

        const result = numbers.filter((item, index, self) => {
            return item % 2 == 1;
        })
        console.log(`result: ${result}`);
        // result: 1,9,25

        const sum = numbers.reduce((previousValue, currentValue, currentIndex, self) => {
            return previousValue + currentValue;
        })
        console.log(`sum: ${sum}`);
        // sum: 55
